import java.util.*;
public class Main
{
	public static void main(String[] args) {
		System.out.println("Hello World");
		Scanner sc=new Scanner(System.in);
		int n=Integer.parseInt(sc.nextLine());
		int[] arr=new int[n];
		 arr[0]=0;
	     arr[1]=1;
		
		for(int i=2; i<n; i++)
		{
		    arr[i]=arr[i-2]+ arr[i-1];
		}
		
		for(int i=0; i<n; i++)
		{
		    System.out.print(arr[i]+" ");
		}
		
		
	}
}
